# Estructura lógica en Python para la Fase 1
from pydantic import BaseModel, Field
from typing import Optional
from openai import OpenAI

class IntencionTurno(BaseModel):
    intencion: str = Field(description="Valores: CONSULTAR, AGENDAR, CANCELAR, OTRAS")
    fecha: Optional[str] = Field(None, description="Fecha extraída en formato AAAA-MM-DD")
    hora: Optional[str] = Field(None, description="Hora en formato HH:MM")
    nombre_cliente: Optional[str] = Field(None, description="Nombre del cliente si lo dice")
    motivo_consulta: Optional[str] = Field(None)

# La llamada estructurada usando el modelo
client = OpenAI()
completion = client.beta.chat.completions.parse(
    model="gpt-4o-mini",
    messages=[
        {"role": "system", "content": "Sos la secretaria. Analizá los mensajes de agenda."},
        {"role": "user", "content": "Quiero reservar el viernes a las 14:00 para hacerme las manos. Me llamo María."}
    ],
    response_format=IntencionTurno,
)
print(completion.choices[0].message.parsed)

# Estructura de escritura en Google Calendar
from googleapiclient.discovery import build
from google.oauth2.credentials import Credentials

def reservar_turno(fecha_iso, hora_iso, cliente_nombre, cliente_tel):
    creds = Credentials.from_authorized_user_file('token.json')
    service = build('calendar', 'v3', credentials=creds)
    
    start_time = f"{fecha_iso}T{hora_iso}:00-03:00"  # Huso horario Argentina
    end_time = f"{fecha_iso}T{int(hora_iso.split(':')[0])+1}:{hora_iso.split(':')[1]}:00-03:00" # 1h despues
    
    event = {
        'summary': f'Turno: {cliente_nombre}',
        'description': f'Registrado por Bot Secretario. Tel: {cliente_tel}',
        'start': {'dateTime': start_time, 'timeZone': 'America/Argentina/Buenos_Aires'},
        'end': {'dateTime': end_time, 'timeZone': 'America/Argentina/Buenos_Aires'},
    }
    
    event_result = service.events().insert(calendarId='primary', body=event).execute()
    return event_result.get('htmlLink')

# Servidor básico FastAPI preparado para recibir Webhooks
from fastapi import FastAPI, Form
import uvicorn

app = FastAPI()

# Memoria temporal de chat en memoria
MEMORIA_CONVERSACION = {}

@app.post("/webhook")
async def receive_message(From: str = Form(...), Body: str = Form(...)):
    print(f"Mensaje de {From}: {Body}")
    
    # Cargar o inicializar historial
    if From not in MEMORIA_CONVERSACION:
        MEMORIA_CONVERSACION[From] = []
        
    MEMORIA_CONVERSACION[From].append({"role": "user", "content": Body})
    
    # Procesar lógica de respuesta y retornar
    return {"status": "success"}

if __name__ == "__main__":
    uvicorn.run(app, host="0.0.0.0", port=8000)

# Envío de mensajes de respuesta usando Twilio
from twilio.rest import Client
import os

def enviar_whatsapp(destinatario, mensaje_texto):
    account_sid = os.environ.get('TWILIO_ACCOUNT_SID')
    auth_token = os.environ.get('TWILIO_AUTH_TOKEN')
    
    client = Client(account_sid, auth_token)
    
    message = client.messages.create(
        from_='whatsapp:+14155238886', # Número de Sandbox de Twilio
        body=mensaje_texto,
        to=destinatario # ej: 'whatsapp:+549342XXXXXXX'
    )
    return message.sid
